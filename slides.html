Python Code


from dataclasses import dataclass
from typing import list
import json

@dataclass
class Slide:
    title: str
    content: list[str]

    def to_html(self) -> str:
        items = "".join(f"<li>{c}</li>" for c in self.content)
        return f"<section><h2>{self.title}</h2><ul>{items}</ul></section>"

JavaScript / TypeScript


interface Config {
  theme: 'light' | 'dark';
  transition: 'slide' | 'fade' | 'none';
  slideNumber: boolean;
}

const initPresentation = (config: Config): void => {
  Reveal.initialize({
    hash: true,
    center: false,
    transition: config.transition,
    slideNumber: config.slideNumber,
  });
};

SQL Query


SELECT
    u.name,
    COUNT(o.id) AS total_orders,
    SUM(o.amount) AS total_spent
FROM users u
JOIN orders o ON o.user_id = u.id
WHERE o.created_at >= '2025-01-01'
GROUP BY u.name
HAVING total_spent > 1000
ORDER BY total_spent DESC
LIMIT 10;

Speaker Notes

Press S to open the speaker view.

You'll see:

  • Elapsed time & clock
These are speaker notes! Only visible in speaker view. You can write detailed talking points here that your audience won't see. Great for rehearsal and live presenting.
